//20190807
#include<stdio.h>
#define inf 9999999
int book[101],sum,n,e[101][101],start;
int data[101],size;

void dfs (int cur){
    book[cur] = 1;
    //printf("%d ",cur);
    data[size++] = cur;
    sum++;
    if(sum==n) return ;
    for(int i=1;i<=n;i++)
        if(e[cur][i]==1&&book[i]==0) 
            dfs(i);
    return ;
}

int main(){
    int i,j,m,a,b;
    scanf("%d %d %d",&n,&m,&start);
    for(i=1;i<=n;i++){
        for(j=1;j<=n;j++){
            if(i==j) e[i][j]=0;
            else e[i][j]=inf;
        }
    }
    
    for(i=1;i<=m;i++){
        scanf("%d %d",&a,&b);
        e[a][b]=1;
        e[b][a]=1;
    }
    dfs(start);
    for(int i=0;i<size;i++){
        printf("%d ",data[i]);
    }
    return 0;
}

//20191204
#include<iostream>
#include<stack>
#define inf 999999
//此题和递归思想高度契合，因此用递归代码更简洁。为加深理解，此处用栈实现 
using namespace std;
int n,m,a,b,cur,book[101],e[101][101];
stack<int> s;
void dfs(int cur){
    if(!book[cur]) cout<<cur<<" ";
    book[cur]=1;
    for(int i=1;i<=n;i++){
        if(e[cur][i]==1&&book[i]==0){
            s.push(i);
            break;
        }
        if(i==n&!(e[cur][i]==1&&book[i]==0)) s.pop();
        //当这个结点去dfs却没有向stack里面添加新元素时，则把这个结点pop 
    }
}

int main(){
    cin>>n>>m;//n vertices, m edges
    for(int i=1;i<=n;i++){
        for(int j=1;j<=n;j++){
            if(i==j) e[i][j]=0;
            else e[i][j]=inf;
        }
    }
    for(int i=1;i<=m;i++){
        cin>>a>>b;
        e[a][b]=1;
        e[b][a]=1;
    }
    
    cur = 2;//start from vertex 2
    s.push(cur);

    while(!s.empty()){
        cur = s.top();
        dfs(cur);
    }
    return 0;
}

//based on queue, view the notes for pseuodo code
#include<iostream>
#include<queue>
#define inf 999999
using namespace std;

int main(){
    int n,m,a,b,cur,book[101]={0},e[101][101],start;
    queue<int> q;
    cin>>n>>m>>start;//n vertices, m edges,start from vertxt start
    for(int i=1;i<=n;i++){
        for(int j=1;j<=n;j++){
            if(i==j) e[i][j]=0;
            else e[i][j]=inf;
        }
    }
    for(int i=1;i<=m;i++){
        cin>>a>>b;
        e[a][b]=1;
        e[b][a]=1;
    }
    cur = start;//start from vertex cur
    q.push(cur);
    book[cur]=1;
    while(!q.empty()){
        cur = q.front();
        cout<<cur<<" ";
        q.pop();
        for(int i=1;i<=n;i++){
            if(e[cur][i]==1&&book[i]==0){
                q.push(i);
                book[i]=1;
            }
        }    
    }
    
    return 0;
}

int chess[6][6];
int count;
int n,m;//bound
int startx,starty,endx,endy;

int dy[4]={1,-1,0,0};
void dfs(int x,int y){    
    if(x == endx && y == endy){
        count++;
        return ;
    }
    for(int i=0;i<4;i++){
        if(chess[x+dx[i]][y+dy[i]]==0 && x+dx[i]>0 && x+dx[i]<=m && y+dy[i]>0 && y+dy[i]<=n){
            chess[x][y] = 1;
            dfs(x+dx[i],y+dy[i]);

        }
    }
}

        scanf("%d %d",&t1,&t2);
        chess[t1][t2] = 2;
    }

    printf("%d\n",count);

//20200204
#include<iostream>
#include<cstdio>
#include<string>
#include<queue>
using namespace std;
bool chess[1010][1010]; //棋盘的原始信息 
int mark[1010][1010]; //存图上每一个点的count值 
bool visited[1010][1010]; //bfs时判断该点是否已经被访问过 
struct point{
    int x;
    int y;
}start,cur;
queue<point> q,q2;
int dx[] = {0,0,1,-1};
int dy[] = {1,-1,0,0};
int main(){
    int n,m;
    scanf("%d %d\n",&n,&m);
    
    for(int i=1;i<=n;i++){
        string t;
        getline(cin,t);
        for(int j=1;j<=n;j++){
            chess[i][j] = t[j-1]-'0';
        }
    }
    for(int k=1;k<=n;k++){
        for(int m=1;m<=n;m++){
            if(visited[k][m]) continue;
            else{
                visited[k][m] = 1;
                point ttt;
                ttt.x = k; ttt.y = m;
                q.push(ttt);
                int count = 1;
                while(!q.empty()){
                    cur = q.front();
                    q2.push(cur);    
                    q.pop();
                    for(int j=0;j<4;j++){
                        if(cur.x+dx[j]>0 && cur.y+dy[j]>0 && cur.x+dx[j]<=n && cur.y+dy[j]<=n &&!visited[cur.x+dx[j]][cur.y+dy[j]]){
                            if(chess[cur.x][cur.y] != chess[cur.x+dx[j]][cur.y+dy[j]]){
                                point temp;
                                temp.x = cur.x+dx[j];
                                temp.y = cur.y+dy[j];
                                q.push(temp);
                                count++;
                                visited[temp.x][temp.y] = 1;
                            }    
                        }    
                    }    
                }
                while(!q2.empty()){
                    mark[q2.front().x][q2.front().y] = count;
                    q2.pop();
                }
                
            }
            
        }
    }
    
    for(int i=0;i<m;i++){
        scanf("%d %d",&start.x,&start.y);
        printf("%d\n",mark[start.x][start.y]);
    }
    
    return 0;
}

//20190820
#include<stdio.h>
#define inf 999999
typedef struct{
    int head;
    int tail;
}queue;
typedef struct{
    int number;
    int times;
}point;

int main(){
    int n,m,a,b,cur,book[101]={0},e[101][101],flag=0;
    int start,end;
    scanf("%d %d %d %d",&n,&m,&start,&end);
    for(int i=1;i<=n;i++){
        for(int j=1;j<=n;j++){
            if(i==j) e[i][j]=0;
            else e[i][j]=inf;
        }
    }
    for(int i=1;i<=m;i++){
        scanf("%d %d",&a,&b);
        e[a][b]=1;
        e[b][a]=1;
    }
    queue q;
    q.head=1;
    q.tail=1;
    
    point p[101];
    p[q.tail].number=start;
    p[q.tail].times=0;
    q.tail++;
    
    book[1]=start;
    while(q.head<q.tail){
        cur=p[q.head].number;
        for(int i=1;i<=n;i++){
            if(e[cur][i]==1&&book[i]==0){
                p[q.tail].number=i;
                p[q.tail].times += 1;
                book[i]=1;
                
            }
            if(p[q.tail].times==end){
                flag=1;
                break;
            }    
        }
        if(flag) break;
        q.head++;
    }
    printf("%d",p[q.tail-1].times);
    
    return 0;
}

//三位数+三位数=三位数，这九位数要求数字不同 
#include<stdio.h>
int a[15],book[15],sum;
int n=9;
void dfs(int step){
    for(int i=1;i<=n;i++){
        if(book[i]==0){
            a[step]=i;
            book[i]=1;
            dfs(step+1);
            book[i]=0;
        }
    }
    if(step==n+1){
        int s1=a[1]*100+a[2]*10+a[3];
        int s2=a[4]*100+a[5]*10+a[6];
        int s3=a[7]*100+a[8]*10+a[9];
        if(s1+s2==s3) {
            printf("%d%d%d+%d%d%d=%d%d%d\n",a[1],a[2],a[3],a[4],a[5],a[6],a[7],a[8],a[9]);
            sum++;
        }
        return ;
    }
    return;
}

int main(){
    dfs(1); 
    printf("%d\n",sum/2);
    
    return 0;
}
