寻找对称日

Feb 2, 2020 2020-02-02T19:32:00+08:00 by Tour
2 min

需求：寻找指定时间段的对称日，对称日所在年份为闰年的需做标记指出

关键点：

　　1.把一年内的所有合法日期以四位数输出（存到数组中）（从1.1到12.31）

　　2.判断是否为闰年

思路：把所有合法日期转化成四位的字符串；

　　把四位的字符串对称扩展成八位，并保存到容器中；

　　按年份从小到大对容器排序；

　　把前四位转化成年份，用来判断该对称日所在年份是否为闰年

//20200202
//range : start from 10000101 , end with 99991231 （所有能用八位数表示的日期）
#include<iostream>
#include<utility>
#include<string>
#include<vector>
#include<algorithm>
using namespace std;
pair<int,int> p[370]; //first=mouth;second=day
char arr[370][8];
vector<string> s;
string start,endw;
bool leap(int y){ //判断闰年
    if( (y%4==0 && y%100!=0) || y%400==0) return true;
    else return false;
}
int main(){
    cin>>start>>endw; //输入起始年份（可输入日期） 
    int k;
    for(int i=1;i<=12;i++){
        if(i==1||i==3||i==5||i==7||i==8||i==10||i==12){
            for(int j=1;j<=31;j++){
                p[k].first = i;    
                p[k++].second = j;
            }        
        }
        else if(i==2){
            for(int j=1;j<=29;j++){
                p[k].first = i;    
                p[k++].second = j;
            }
        }
        else{
            for(int j=1;j<=30;j++){
                p[k].first = i;    
                p[k++].second = j;
            }
        }
    }
    for(int i=0;i<k;i++){
        arr[i][3] = arr[i][4] = p[i].first/10+'0';
        arr[i][2] = arr[i][5] = p[i].first%10+'0';
        arr[i][1] = arr[i][6] = p[i].second/10+'0';
        arr[i][0] = arr[i][7] = p[i].second%10+'0';
        if(arr[i]>=start && arr[i]<=endw) {
            s.push_back(arr[i]);
        }
    }
    sort(s.begin(),s.end());
    cout<<"count = "<<s.size()<<endl;
    vector<string>::iterator it = s.begin();
    for(it;it!=s.end();it++){
        string temp = *it;
        int year = (temp[0]-'0')*1000 + (temp[1]-'0')*100 + (temp[2]-'0')*10 + temp[3]-'0';
        if(leap(year)) cout<<*it<<" "<<"leap year"<<endl;
        else cout<<*it<<endl;
    }
    return 0;
} 
//输出从2000到2099年所有对称日
